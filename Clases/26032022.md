### Metodos

```java
public class Main {

    // Declarar un metodo
    // <Tipo> <Nombre>(<Parametros>)
    // {
    //    <cuerpo>
    // }
    // Como declarar un tipo de metodo?
    // public static <tipo de valor retorno>
    //                         void -> no hay un dato de retorno
    //                     int, double, String, boolean, ...
    public static void metodo() {
        // Alguna logica simple
        System.out.println("Hola");
    }

    public static int calcularMonedas5Soles(int monto) {
        int cantidadMonedas5 = monto / 5;
        return cantidadMonedas5;
    }

    public static void calcularCantidadDeMonedas(int monto) {
        int valorMoneda5 , restoMoneda5;
        int valorMoneda2 , restoMoneda2;
        int valorMoneda1 , restoMoneda1;

        //Evaluamos cuantas monedas de 5 y su resto
        valorMoneda5 = calcularMonedas5Soles(monto);
        restoMoneda5 = monto%5;

        //Evaluamos cuantas monedas de 2 y su resto
        valorMoneda2 = restoMoneda5/2;
        restoMoneda2 = valorMoneda2%2;

        //Evaluamos cuantas monedas de 1 y su resto
        valorMoneda1=restoMoneda2/1;
        restoMoneda1=valorMoneda1%1;

        System.out.println("Se necesitan " + valorMoneda5 +" de Monedas de 5 soles");
        System.out.println("Se necesitan " + valorMoneda2 +" de Monedas de 2 soles");
        System.out.println("Se necesitan " + valorMoneda1 +" de Monedas de 1 soles");
        System.out.println("======");
    }


    public static void main(String[] args) {

        calcularCantidadDeMonedas(23);
        calcularCantidadDeMonedas(52);
        calcularCantidadDeMonedas(25);

        int resultado1 = calcularMonedas5Soles(23);
        System.out.println("Cantidad de monedas de 5 soles = " + resultado1);
        int resultado2 = calcularMonedas5Soles(67);
        System.out.println("Cantidad de monedas de 5 soles = " + resultado2);

        // 13 + 4 = 17
        int resultadoGlobal = resultado1 + resultado2;
        System.out.println(resultadoGlobal);
    }
}

```
